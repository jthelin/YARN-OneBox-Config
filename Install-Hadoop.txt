set JAVA_HOME
set HADOOP_YARN_HOME env variable

Add $HADOOP_YARN_HOME/bin and $HADOOP_YARN_HOME/sbin to $PATH


Set up ssh for local access. Test with ssh localhost

ssh-keygen -t rsa -f ~/.ssh/id_rsa
cat ~/.ssh/id_rsa.pub >> ~/.ssh/authorized_keys
ssh localhost


Format HDFS file system

hadoop namenode -format


Start daemon processes

start-dfs.cmd
start-yarn.cmd


<do stuff>


Stop daemon processes

stop-dfs.cmd
stop-yarn.cmd
